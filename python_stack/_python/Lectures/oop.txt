class Person:
    #Attributes(or characteristics)
    def __init__(self, name, age, limbs, hair_color): #parameters to set up object
        print('a new human is born')
        self.name = name
        self.age = age
        self.limbs = limbs
        self.hair_color = hair_color
    #methods
    def say_hello(self):
        print('Hello I am awesome!')
        print(f'Hello, my name is {self.name} and I am {self.age} years old.)
    def birthday(self):
        self.age+=1
        print(f'oh wow rick, look at me, I'm {self.age}')


#Instantiating a class
    person1 = Person('Bob', 43, 4, 'Red')
    person2 = Person('Mort', 14, 4, 'Pink')

    person1.say_hello()
    person2.say_hello()

    person2.birthday()
